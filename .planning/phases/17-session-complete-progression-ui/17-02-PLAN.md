const threshold3 = getFlag('chessStarThreshold3');

   (Initialize once on mount — no useEffect needed)

4. Tier advancement detection — iterate chessPieces (excluding pawn — pawn has no puzzles in session queue per Phase 16 decision which only uses first 5 pieces sorted by order):
```typescript
const advancedPieces: string[] = [];
const piecesForDisplay = chessPieces.filter(p => p.id !== 'pawn');
for (const piece of piecesForDisplay) {
  const startTier = sessionTiers.current[piece.id] ?? 1;
  const currentTier = currentTiersByPiece[piece.id]?.tier ?? 1;
  if (currentTier > startTier) {
    advancedPieces.push(piece.id);
  }
}
```

5. Band name helper:
```typescript
function getBandKey(tier: 1 | 2 | 3): string {
  if (tier === 3) return 'ui.masteryExpert';
  if (tier === 2) return 'ui.masteryIntermediate';
  return 'ui.masteryBeginner';
}
```

6. Tier color helper (matching LEVELS array colors from ChessGameContent):
```typescript
function getTierColor(tier: 1 | 2 | 3): string {
  if (tier === 3) return '#ffcd36'; // gold
  if (tier === 2) return '#dbc3e2'; // purple
  return '#9ed6ea'; // blue
}
```
