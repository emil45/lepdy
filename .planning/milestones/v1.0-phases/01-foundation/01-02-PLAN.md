**1. Movement Puzzle type:**
```typescript
import { ChessPieceId } from './chessPieces';

export interface MovementPuzzle {
  id: string;                    // e.g. 'rook-move-1'
  pieceId: ChessPieceId;         // which piece is being taught
  fen: string;                   // FEN position (only piece placement part, e.g. '8/8/8/4R3/8/8/8/8')
  pieceSquare: string;           // algebraic notation of the piece, e.g. 'e5'
  validTargets: string[];        // all legal destination squares
  difficulty: 1 | 2 | 3;        // 1=easy (few options), 2=medium, 3=harder
}
```

**3. Capture Puzzle type:**
```typescript
export interface CapturePuzzle {

}
```
